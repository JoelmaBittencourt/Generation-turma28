--inciar  server zookeeper -  fica centralizada a estrutura de tópicos e partiçoes. ferramenta otimizada para armazernar chave e valores.


kafka-server-start.bat C:\kafka\config\server.properties -

-- inciar o kafka;



-listar topicos 

kafka-topics --bootstrap-server localhost:9092 --list

-criar topico

kafka-topics --bootstrap-server localhost:9092 --create --topic "nome do tipo" sem aspas.(caso queira colocar o numero de partições acrecente --partitions "quantidade")


- Inserir dados via linha de comando "produção"

kafka-console-producer.bat --broker-list localhost:9092 --topic "nome do topico"

- Consumir as mensagens 

kafka-console-consumer --bootstrap-server localhost:9092 --topic "nome do topico" [--from-beginning(adicionar para cosnumir desde o começo)] ou [-- group "nome do grupo]


- informar os dados do topico 

kafka-topics --bootstraps-server localhost:9092 --topic "nome do topico" --describe

-criar novas partições

kafka-topics --bootstrap-server localhost:9092 -- alter --topic "nome do topico" --partitions "quantidade de partições"


-exclusão de topico

kafka-topics --bootstrap-server localhost:9092 --delete --topic "nome do topico"

--chave e separador

kafka-console-producer --broker-list 127.0.0.1:9092 --topic teste --property "parse.key=true" --property "key.separator=,"

--consumo chave e separador

kafka-console-consumer --bootstrap-server 127.0.0.1:localhost:9092 --topic teste --property "print.key=true" --property "key.separator=," --group grupo1
